int q;
vii prefix_function( string pat ){
    vii pre; pre.resize(pat.L , 0);
    for ( int i = 1 , k = 0; i < (int)pat.L; ++i ){
        while ( k > 0 && pat[i] != pat[k] ){
            k = pre[k - 1];
        }
        if ( pat[i] == pat[k] ) ++k;
        pre[i] = k;
    }
    return pre;
}

int main(){
    FASTIO //ONLINE__JUDGE();

    cin >> q;
    while ( q-- ){
        string s; cin >> s;
        vii pre = prefix_function( s );
        int len = s.L - pre.back();

        if ( s.L%len == 0 && len <= s.L/2 ){
            cout << len << "\n";
        }
        else{
            cout << s.L << "\n";
        }

        cout << (q > 0 ? "\n" : "");
    }
    return 0;
}
