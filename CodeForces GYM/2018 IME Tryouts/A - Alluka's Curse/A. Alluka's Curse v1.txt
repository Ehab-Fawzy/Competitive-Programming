const ll MOD = 1e9 + 7;
const ll MAX = 1e2 + 7;
 
const int N = 1e7 + 9;
unsigned int dp[N][5]; ll in;
 
int main()
{
    FASTIO //ONLINE__JUDGE();
 
    cin >> in;
    vector<int> idx(7 , 0);
    idx[0] = 0; idx[1] = 1;
    idx[3] = 2; idx[4] = 3;
    idx[6] = 4;
 
    dp[in][0] = 1;
    for ( int i = in; i >= 0; --i ){
        for ( int msk : {0,1,2,3,4} ){
            if      ( msk == 0 ){
                if (i+1 < in) dp[i][msk] += dp[i + 2][idx[0]];
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[1]];
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[4]];
                dp[i][msk] %= MOD;
            }
            else if ( msk == 1 ){
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[6]];
                dp[i][msk] += dp[i + 1][idx[0]] , dp[i][msk] %= MOD;
            }
            else if ( msk == 2 ){
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[4]] , dp[i][msk] %= MOD;
            }
            else if ( msk == 3 ){
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[3]];
                dp[i][msk] += dp[i + 1][idx[0]] , dp[i][msk] %= MOD;
            }
            else if ( msk == 4 ){
                if (i+1 < in) dp[i][msk] += dp[i + 1][idx[1]] , dp[i][msk] %= MOD;
            }
        }
    }
 
    ll ret = dp[2][idx[0]] + dp[1][idx[1]] + dp[1][idx[4]]; ret %= MOD;
    cout << ret;
    return 0;
}
