const ll MAX = 1e5 + 7;

int p[MAX] , r[MAX];

void init( int x ){
    for ( int i = 0 ; i <= x; ++i ){
        p[i] = i; r[i] = 0;
    }
}
int root( int i ){
    if ( i == p[i] ) return i;
    return p[i] = root(p[i]);
}
void sum( int i , int j ){
    int x = root(i) , y = root(j);
    if ( x != y ){
        if ( r[x] > r[y] ) swap(x,y);
        p[x] = y; r[x] == r[y] ? r[y]++ : 0;
    }
}
